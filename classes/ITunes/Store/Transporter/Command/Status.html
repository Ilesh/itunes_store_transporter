def handle_success(stdout_lines, stderr_lines, options)
  # Pre-XML behavior. Not sure if it should be kept.
  return [] if stdout_lines.empty?

  begin
    XML::Status.new.parse(stdout_lines.join(""))
  rescue ParseError => e
    raise TransporterError, e.message
  end
end
